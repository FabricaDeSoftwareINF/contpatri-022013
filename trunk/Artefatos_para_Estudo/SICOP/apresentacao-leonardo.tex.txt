\subsection{Documento Visão}
\frame
{
	\frametitle{Documento Visão}
    \framesubtitle{Definição do problema}
	\begin{footnotesize}
	\begin{tabular}{|p{3cm}|p{7cm}|}
		\hline
			\textbf{O problema de} & Não ter um sistema adequado, dificulta o controle administrativo e financeiro de patrimônio.\\
		\hline
			\textbf{Afeta} & Funcionários e administração.\\
		\hline
			\textbf{Cujo impacto é} & Má organização, má administração, mau controle e aproveitamento.\\
		\hline
			\textbf{Uma solução de sucesso seria} & Um sistema adequando para resolver estas necessidades/falhas à fim de alcançar os objetivos almejados, que são: controle e gerenciamento de patrimônio, estatísticas para controle estratégico e aprimoramento das informações administrativas.\\
		\hline
	\end{tabular}
	\end{footnotesize}
}

\frame
{
	\frametitle{Documento Visão}
	\framesubtitle{Sentenças de Posicionamento do Produto}
	\begin{footnotesize}
	\begin{tabular}{|p{4cm}|p{6cm}|}
		\hline
			\textbf{Para} & Faculdades ou Universidades.\\
		\hline
			\textbf{O Sistema de Gerenciamento de Patrimônio} & É um sistemas computacional de controle e gerenciamento de patrimônio.\\
		\hline
			\textbf{Que} & Organiza, controla, gerencia os bens patrimoniais da empresa gerando informações estratégicas, buscando dar um visão macro do passivo, bem como suas depreciações, localizações e outras informações relavantes ao adminstrador.\\
		\hline
			\textbf{Diferente de} & Produtos que não informam adequadamente aos administradores as necessidades atuais, com ferramentas ultrapassadas que geram confusão e dificultam o trabalho\ldots \\
		\hline
			\textbf{Nosso produto} & Qualidade, sofisticação adotando padrões de alta tecnologia, precisão nas informações geridas, rapidez das buscas ao sistema\ldots\\
		\hline
	\end{tabular}
	\end{footnotesize}
}

\frame
{
	\frametitle{Documento Visão}
	\framesubtitle{Necessidades dos Usuários}
	\begin{footnotesize}
	\begin{tabular}{|p{2cm}|p{8cm}|}
		\hline
			\textbf{Descrição} & Gerir informações sobre o patrimônio da empresa em um sistema de informação gerecial.\\
		\hline
			\textbf{Solução Atual} & Sistema lento, sem manutenção e descontinuado.\\
		\hline
			\textbf{Solução Proposta} & Sistema rápido, usufruindo de tecnologias inovadoras e usabilidade adequada, com manutenção da aplicação de dase de dados, continuas versões e módulos em desenvolvimento \ldots \\
		\hline
			\textbf{Prioridade} & Rapidez e integridade.\\
		\hline
	\end{tabular}
	\end{footnotesize}
}
